Step 05

同步到 Google Sheet

用 Google Apps Script 建立一個「Web App」網址，工具會把目前篩選出的結果，以條列式附加寫入到你的 Google Sheet，每次執行都是新增一批（含資料撈取時間、觀察區間），方便累積歷史紀錄。

還沒設定過？點這裡看 5 步驟設定教學
  1. 打開你要記錄結果的 Google Sheet，點選選單「擴充功能 → Apps Script」
  2. 把預設程式碼全部刪除，貼上下面這段程式碼
  3. 點右上角「部署 → 新增部署作業」，類型選「網頁應用程式」
  4. 「執行身分」選你自己，「誰可以存取」選「任何人」，然後部署
  5. 複製產生的「網頁應用程式網址」，貼到下面的欄位
function appendRows_(ss, sheetName, rows) {
  if (!rows || rows.length === 0) return 0;
  var sheet = ss.getSheetByName(sheetName) || ss.insertSheet(sheetName);
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(Object.keys(rows[0]));
  }
  var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
  var lines = rows.map(function (row) {
    return headers.map(function (h) {
      return row[h] !== undefined ? row[h] : '';
    });
  });
  sheet.getRange(sheet.getLastRow() + 1, 1, lines.length, headers.length).setValues(lines);
  return lines.length;
}

function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var payload = JSON.parse(e.postData.contents);

  var addedDetail = appendRows_(ss, '比對紀錄', payload.rows || []);
  var addedSummary = appendRows_(ss, '比對彙總', payload.summary ? [payload.summary] : []);

  return ContentService.createTextOutput(JSON.stringify({
    ok: true,
    added: addedDetail,
    addedSummary: addedSummary
  })).setMimeType(ContentService.MimeType.JSON);
}
